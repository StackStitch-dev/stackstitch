    class RunInvestigation:
      def __init__(self, investigation_repo: InvestigationRepository, insight_repo: InsightRepository, invocation_repo: InvocationRepository, investigator: Investigator, event_publisher: EventPublisher) -> None
      async def execute(self, project_id: UUID, trigger: str, trigger_ref: UUID, query: str | None = None) -> Insight | None:
        - Convert trigger string to InvestigationTrigger enum
        - Create Investigation(project_id=project_id, trigger=trigger_enum, trigger_ref=trigger_ref, query=query)
        - investigation.start()
        - try:
            result = await self._investigator.investigate(investigation, context={})
            investigation.complete(result)
            insight = Insight(project_id=project_id, investigation_id=investigation.id, title=result.findings[:100], narrative=result.findings, insight_type=InsightType.ANOMALY_EXPLANATION if trigger_enum == InvestigationTrigger.ANOMALY else InsightType.AD_HOC_RESPONSE)
            await self._investigation_repo.save(investigation)
            await self._insight_repo.save(insight)
            event = InsightCreated(insight_id=insight.id, investigation_id=investigation.id, project_id=project_id)
            await self._event_publisher.publish(event)
            invocation = Invocation(thread_id=trigger_ref, project_id=project_id, source=InvocationSource.INSIGHT, role="system", message=f"New insight: {insight.title}")
            await self._invocation_repo.save(invocation)
            return insight
          except Exception as e:
            investigation.fail(str(e))
            await self._investigation_repo.save(investigation)
            return None

    3. core/src/core/application/use_cases/orchestrate.py per D-30, D-31, D-32, D-66:
    class Orchestrate:
      def __init__(self, thread_repo: ThreadRepository, invocation_repo: InvocationRepository, agent: Agent, message_deliverer: MessageDeliverer) -> None
      async def execute(self, thread_id: UUID) -> None:
        - thread = await self._thread_repo.get_by_id(thread_id)
        - If None, return early (nothing to orchestrate)
        - last_response: str | None = None
        - DRAIN LOOP:
          while True:
            pending = await self._invocation_repo.get_pending_by_thread_id(thread_id)
            if not pending:
              break
            for inv in pending:
              inv.mark_processing()
              await self._invocation_repo.save(inv)
            response = await self._agent.process(thread, pending)
            last_response = response
            # Store intermediate response in thread per D-31
            assistant_msg = Message(role=MessageRole.ASSISTANT, content=response)
            thread.add_message(assistant_msg)
            await self._thread_repo.save(thread)
            for inv in pending:
              inv.mark_done()
              await self._invocation_repo.save(inv)
        - # Deliver only final response per D-32
        - if last_response is not None:
            await self._message_deliverer.deliver(thread_id, last_response)
